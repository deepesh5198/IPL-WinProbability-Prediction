$ git config --global user.name "name"
$ git config --global user.email "emailid"

$ git init

$ git add .

$ git commit -m "any message (for eg. "Initial Commit")"

$ git status

$ git remote add origin "paste SSH link here"

$ git push origin master (if public: no problem else if Private: need to generate an SSH key)

$ git ssh-keygen -b 2048 -t rsa
  # this will generate ssh key at given location
  # copy the location

$ git cat "paste copied location"
  # this shows a long key which is your ssh key, copy it and paste it in the SSH key in github

$ git push origin master 
  # starts pushing your project to github
